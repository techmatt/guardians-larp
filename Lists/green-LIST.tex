\DECLARESUBTYPE{Green}{Element}
\PRESETS{Green}{
  \FD\MYfile	{} %% filename (in \greens dir)
  \FS\MYtext	{\ifx\MYfile\empty\else%
		  \getextractenvs{document}{\greens/\MYfile}%
		\fi}
  }
